# spill_stigespill-funksjonen fra kapittel 7.7

    (1, 40), (8, 10), (24, 5),
    (33, 3), (36, 52), (42, 30),

    (64, 27), (65, 82), (68, 85),

def spill_stigespill(): 
    posisjon = 0
    antall_kast = 0 
    while posisjon < 90:
        terningkast = np.random.randint(1, 7) 
        antall_kast += 1
        posisjon += terningkast
        for stige in stiger:
           if stige[0] == posisjon:
              posisjon = stige[1]
              break;
    return antall_kast

N = 10000
